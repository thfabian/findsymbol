cmake_minimum_required(VERSION 2.8)

################################################################################
# Compilation
################################################################################
file(GLOB sources "*.cpp")
add_executable(findsymbol ${sources})
target_link_libraries(findsymbol FSDriver
                                 FSObjectDump 
                                 FSDemangle
                                 FSSupport
                                 ${LibIberty_LIBRARIES}
                                 ${PUGIXML_LIBRARIES}
                                 ${TBB_LIBRARIES} 
                                 ${LLVM_LIBRARIES}
                                 ${LLVM_LDFLAGS})
                                 
install(TARGETS findsymbol RUNTIME DESTINATION bin)
